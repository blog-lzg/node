1. 数据类型和约束[记忆,理解] #

  • 数据类型
    • 数字类型 int
    • 小数decimal
    • 可变字符串varchar(数字)
    • 固定长度字符串char(数字)
    • 时间date
  • 约束
    • 主键: primary key
    • 非空: not null

添加数据的时候,会对数据的数据类型和约束进行验证,如果不满足要求,则会添加失败, 保证数据的准确性

2. 命令行客户端的使用-登录和登出[掌握] #

想要能够连接的前提: 服务端已经启动了
ps aux | grep mysql
sudo service mysql start

-- 登录(连接数据) 客户端连接服务端 (用户名 root, 密码 mysql)
mysql -uroot -p
-- 登出(退出数据库连接) 关闭客户端和服务端的链接
exit/ quit/ ctrl d
-- 查询时间(登录状态)
select now();

4. 命令行客户端使用-表结构操作SQL语句(掌握) #

-- 数据准备
create database py charset=utf8;

-- 查看数据库中所有的表(已经选择一个数据库)
show tables;

-- 创建数据表
create table 表名(字段 数据类型 数据约束,字段 数据类型 数据约束, ... );
	-- 创建学生表
	-- id 主键 id int unsigned primary key not null auto_increment(自动增长)
	--  姓名 name varchar(20) not null
	--  年龄 age tinyint default 0
	--  身高 height decimal(5,2) 
	--  性别 gender enum('男', '女', '保密')
	create table stu(
        id int unsigned primary key not null auto_increment, 
        name varchar(20) not null, 
        age tinyint default 0, 
        height decimal(5,2), 
        gender enum('男', '女', '保密')
    );

-- 删除表
drop table 表名;
	-- 删除 stu 表
	drop table stu;

5. 命令行客户端使用-表数据的操作(掌握) #

添加数据 #

-- 添加数据 -- 全列插入
-- insert into 表名 values(数据, 数据, 数据, ...)  有多少个字段,就书写多少个数据
	-- id 是主键,设置了自动增长,一般写 0
insert into stu values(0, '张三', 18, 165.3, '男');
insert into stu values(0, '张三', 18, 165.5, '女');
insert into stu values(0, '张三', 18, 165.7, '保密');

-- 对于枚举类型来说,可以使用数字(下标,从 1 开始)代替具体的字符串
insert into stu values(0, '张三丰', 18, 166.7, 1);
-- 添加数据 -- 插入指定列数据
insert into 表名(字段, 字段, ...) values (数据, 数据, ...);
-- 只添加 name 和 age 字段的数据,其他的字段使用默认值,或者 null 值

insert into stu(name, age) values ('李四', 19);
insert into stu(name, height) values ('王五', 170.1);

-- 添加数据 -- 插入多行数据
insert into 表名(字段, 字段,...) values (), (), ();
insert into stu(name, height) values('张飞', 170), ('关于', 180), ('刘皇叔', 175), ('赵云', 176);

比较运算符查询 #

-- > 大于 < 小于 >= 大于等于 <= 小于等于 = 等于  != 不等于  <>  不等于

-- 查询编号大于3的学生
select * from stu where id > 3;
-- 查询编号不大于4的学生
select * from stu where id <=4;

-- 查询姓名不是“张三”的学生
select * from stu where name != '张三';

逻辑运算符查询 #

-- and or not 

--查询编号大于3的女同学
select * from stu where id>3 and gender='女';

--查询编号小于4或没被删除的学生
select * from stu where id <4 or gender='女';

--查询年龄不在10岁到15岁之间的学生
select * from stu where  age < 10 or age >15;

select * from stu where not (age >=10 and age <=15);

模糊查询 #

-- 使用关键字 like , % 任意多个字符, _ 任意一个字符
select * from 表名 where 字段 like '';

-- 查询姓张的学生 张%
select * from stu where name like '张%';

--查询姓张并且“名”是一个字的学生 张_
select * from stu where name like '张_';

--名字是两个字的学生 '__'
select * from stu where name like '__';

--查询姓张或叫三的学生 "张%"  or  "%三"
select * from stu where name like '张%' or name like '%三';

范围查询 #

-- between A and  B   连续的范围
-- in (a, b, c)   在里边的内容

-- 查询编号为3至7的学生
select * from stu where id between 3 and 7;

--查询编号是4、5、7的学生
select * from stu where id=4 or id = 5 or id =7;
select * from stu where id in (4, 5, 7);
--查询编号不是4、5、7的学生
select * from stu where id not in (4, 5, 7);

10. 聚合函数(组函数)[掌握] #

-- 统计个数,count()
select count(height) from stu;
-- 聚合函数, 默认不会统计 null 数据
-- 想要统计全部数据可以使用不为空的列来统计(id)
select count(id) from stu;

-- 最小值,min() , 
	-- 查询最小的年龄
	select min(age) from stu;
	
-- 最大值max()
	-- 查询最大的年龄	
	select max(age) from stu;
	
-- 求和 sum()
	select sum(height) from stu;
-- 平均数 avg()
select avg(height) from stu;
